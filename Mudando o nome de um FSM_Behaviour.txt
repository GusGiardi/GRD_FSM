Para mudar o nome de um FSM_Behaviour, adicione o atributo MovedFromAttribute passando o Assembly, Namespace e ClassName originais.

Exemplo
using UnityEngine.Scripting.APIUpdating;

namespace New.Namespace
{
	[MovedFrom(true, sourceAssembly: "Assembly-CSharp", sourceNamespace:"Old.Namespace", sourceClassName: "Old_ClassName")]
	public class New_ClassName : FSM_StateBehaviour
	{
		...
	}
}
